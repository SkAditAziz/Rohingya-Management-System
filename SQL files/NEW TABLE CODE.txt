CREATE TABLE EMPLOYEES (
  EMPLOYEE_ID NUMBER NOT NULL PRIMARY KEY,
  CAMP_ID NUMBER REFERENCES,
  E_FNAME VARCHAR(255),
  E_LNAME VARCHAR(255),
  E_CONTACT_NO NUMBER,
  E_SALARY NUMBER,
  E_JOB_ID VARCHAR(255)
);

CREATE TABLE CAMP (
  CAMP_ID NUMBER NOT NULL PRIMARY KEY,
  C_LOCATION VARCHAR(255),
  C_CAPACITY NUMBER,
  C_MANAGER_ID NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)
);


CREATE TABLE ROHINGYA (
  ROHINGYA_ID NUMBER NOT NULL PRIMARY KEY,
  CAMP_ID NUMBER REFERENCES CAMP(CAMP_ID),
  R_FNAME VARCHAR(255),
  R_LNAME VARCHAR(255),
  R_AGE NUMBER,
  R_GENDER VARCHAR(10),
  R_ADDRESS VARCHAR(255),
  R_GUARDIAN_ID NUMBER REFERENCES ROHINGYA(ROHINGYA_ID),
  R_ENTERING_DATE DATE,
  R_EMPLOYEE_ID NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)

);

CREATE TABLE PREGNANCY (
  ROHINGYA_ID NUMBER NOT NULL PRIMARY KEY REFERENCES ROHINGYA(ROHINGYA_ID),
  PREG_START_DATE DATE,
  DOCTORS_ID NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)
);

CREATE TABLE DEATH (
  ROHINGYA_ID NUMBER NOT NULL PRIMARY KEY REFERENCES ROHINGYA(ROHINGYA_ID),
  DATE_OF_DEATH DATE,
  ENTRIED_BY NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)
);

CREATE TABLE NEW_BORN (
  NEW_BORN_ID NUMBER NOT NULL PRIMARY KEY,
  MOTHERS_ID NUMBER REFERENCES ROHINGYA(ROHINGYA_ID),
  DATE_OF_BIRTH DATE,
  DOCTORS_ID NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)
);


CREATE TABLE DISEASE (
  DISEASE_ID NUMBER NOT NULL PRIMARY KEY,
  D_NAME VARCHAR(255),
  D_TYPE VARCHAR(255),
  D_DEPARTMENT VARCHAR(255)
);

CREATE TABLE DISEASE_DETECTION (
  DISEASE_ID NUMBER REFERENCES DISEASE(DISEASE_ID),
  DOCTORS_ID NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID),
  ROHINGYA_ID NUMBER REFERENCES ROHINGYA(ROHINGYA_ID),
  DATE_OF_DETECTION DATE,

  CONSTRAINT DISEASE_DETECTION_PK PRIMARY KEY (DISEASE_ID,DOCTORS_ID,ROHINGYA_ID)
);


CREATE TABLE RELIEF_ITEM (
  ITEM_ID NUMBER NOT NULL PRIMARY KEY,
  ITEM_NAME VARCHAR(255),
  ITEM_TYPE VARCHAR(255),
  ITEM_COST_PER_UNIT NUMBER
);

CREATE TABLE RELIEF_STORAGE (
  ITEM_ID NUMBER NOT NULL PRIMARY KEY REFERENCES RELIEF_ITEM (ITEM_ID),
  ITEM_UNIT NUMBER (8,2)
);

CREATE TABLE RELIEF (
  RELIEF_SERIAL_NO NUMBER NOT NULL PRIMARY KEY,
  ROHINGYA_ID NUMBER REFERENCES ROHINGYA(ROHINGYA_ID),
  RELIRF_GIVEN_TIME DATE,
  RELIEF_GIVEN_BY NUMBER REFERENCES EMPLOYEES(EMPLOYEE_ID)
);


CREATE TABLE RELIEF_AMOUNT (
  RELIEF_SERIAL_NO NUMBER NOT NULL,
  ITEM_ID NUMBER,
  ITEM_AMOUNT NUMBER (5,2),
  CONSTRAINT RELIEF_AMOUNT_PK PRIMARY KEY (RELIEF_SERIAL_NO,ITEM_ID)
);
